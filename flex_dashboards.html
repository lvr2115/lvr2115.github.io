weather_data = 
  ny_noaa |>
  mutate(
    name = recode(
      id, 
      USW00094728 = "CentralPark_NY")) |>
  mutate(tmin=as.numeric(tmin), tmax = as.numeric(tmax)) |>
  mutate(tmin = tmin / 10,
    tmax = tmax / 10) |>
  drop_na(tmin, tmax) |>
  select(name, id, everything()) |>
  filter(name%in% c("CentralPark_NY"))
